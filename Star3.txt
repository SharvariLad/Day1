using System;
public class Program
{
	public static void Main(String[] args)
	{
       
		int n,m=1;
		
		n=3;
	
		for(int i=n;i>=1;i--)  
   		{  
       		for(int j=1;j<m;j++)  
       		{  
           		Console.Write(" ");  
       		}  
       		for(int k=1;k<=2*i-1;k++)  
       		{  
           		Console.Write("*");  
      		 }  
        m++;  
     
     		 Console.WriteLine(" ");  
   		}
		
		Console.ReadLine();
	}
}
